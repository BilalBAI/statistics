Example 2: Random walks

## generate three rank walk time plots
par(mfrow=c(1,3),mai=c(0.3,0.3,0.3,0.1))
for (i in 1:3){
n <- 100; 
e <- rnorm(n); z <- as.numeric(); z[1] <- e[1]
for(i in 2:n){
 z[i]=z[i-1]+e[i]
}
plot(z); lines(z);
}
